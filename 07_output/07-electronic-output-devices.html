Explanation

For my MVP, I decided to tackle the casing and logic of the Mapper. This prototype is equipped with the MakerFocus TF-Luna LiDAR Range Finder Sensor, which creates a 3D point map using spherical coordinates. The other two coordinates (phi and theta) are calculated based on the position of the stepper motor at a given time, resulting in an array of vectors. This array of vectors is then mapped using the Pygame library, creating the final render you see. The actual housing of the sensor was 3D printed (see top for model). I then used the laser cutter to create the casing for the electronics and the second stepper motor.

Arduino Code

const int stepPin = 4;
const int dirPin = 2;

int thetaStep = 0;
const int stepsPerRevolution = 200; // Adjust to your motor's specs
const float thetaMaxDeg = 45.0;
const float degPerStep = thetaMaxDeg / stepsPerRevolution;

void setup() {
  pinMode(stepPin, OUTPUT);
  pinMode(dirPin, OUTPUT);
  Serial.begin(115200);
  digitalWrite(dirPin, LOW); // Adjust direction as needed
}

void loop() {
  // Step angle in degrees
  float theta = thetaStep * degPerStep;

  // Simulated distance (replace with actual sensor reading later)
  float r = 100.0 + 10.0 * sin(thetaStep * 0.1);

  // Print angle and distance
  Serial.print("theta: ");
  Serial.print(theta);
  Serial.print(" r: ");
  Serial.println(r);

  // Wait before stepping
  delay(200); // time to stabilize or read sensor

  // Perform one step
  digitalWrite(stepPin, HIGH);
  delay(2); // pulse duration
  digitalWrite(stepPin, LOW);
  delay(2);

  // Wait after stepping
  delay(300); // additional delay between steps

  // Increment angle step
  thetaStep++;
  if (thetaStep >= stepsPerRevolution) {
    thetaStep = 0;
    delay(1000); // pause at end of sweep
  }
}

import pygame
import serial
import math
import sys

# === Serial Setup ===
SERIAL_PORT = 'COM6'  # Change to your actual port
BAUD_RATE = 115200
try:
    ser = serial.Serial(SERIAL_PORT, BAUD_RATE, timeout=1)
except serial.SerialException as e:
    print(f"Error opening serial port: {e}")
    sys.exit(1)

# === Pygame Setup ===
WIDTH, HEIGHT = 800, 600
CENTER = (WIDTH // 2, HEIGHT // 2)
SCALE_X = 3   # Horizontal zoom
SCALE_Y = 3   # Vertical zoom

pygame.init()
screen = pygame.display.set_mode((WIDTH, HEIGHT))
pygame.display.set_caption("Live Euclidean Plot")
clock = pygame.time.Clock()

BLACK = (0, 0, 0)
RED = (255, 0, 0)

running = True
points = []

frame_counter = 0
frames_per_point = 50  # Add point every 50 frames to slow plotting

while running:
    screen.fill(BLACK)

    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False

    # Only read and plot every 'frames_per_point' frames to slow down
    if frame_counter % frames_per_point == 0:
        if ser.in_waiting:
            try:
                line = ser.readline().decode('utf-8').strip()
                # Expect format: "theta: XX r: YY"
                if line.startswith("theta:"):
                    parts = line.replace("theta:", "").replace("r:", "").split()
                    theta = float(parts[0])
                    r = float(parts[1])

                    theta_rad = math.radians(theta)
                    x = r * math.cos(theta_rad)
                    y = r * math.sin(theta_rad)

                    screen_x = int(CENTER[0] + x * SCALE_X)
                    screen_y = int(CENTER[1] - y * SCALE_Y)

                    points.append((screen_x, screen_y))

            except Exception:
                pass  # ignore bad lines/errors

    frame_counter += 1

    for pt in points:
        pygame.draw.circle(screen, RED, pt, 4)

    pygame.display.flip()
    clock.tick(60)

ser.close()
pygame.quit()
sys.exit()
